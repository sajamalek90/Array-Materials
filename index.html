Array Fundamentals

Array Introduction

Arrays are ordered collections of values that can hold multiple items of any type in a single variable.

// Create Arrays and Access Array Elements
let fruits = ['Apple', 'Banana', 'Orange'];
let mixedData = [1, 'Hello', true, {name: 'John'}, ['Apple', 'Banana', 'Orange']];

console.log(fruits[0]);
console.log(fruits[2]);
console.log(fruits[1][3]);
console.log(fruits[2][0]);

console.log(mixedData[4][0][1]);

console.log(fruits);
fruits[2] = 'Pomegranate';
console.log(fruits);

console.log(mixedData)
mixedData[4] = 'Watermelon';
console.log(mixedData);

Arrays are zero-indexed: the first element is at position 0.

Using Length With Array

The length property returns the number of elements in an array.

let fruits = ['Apple', 'Banana', 'Orange'];
console.log(fruits.length);

fruits[6] = 'Strawberries'; // Add element at the index 6 of the array
console.log(fruits);
console.log(fruits.length); 

fruits[fruits.length] = 'Strawberries'; // Add element after the last index of the array
console.log(fruits);
console.log(fruits.length); 

fruits[fruits.length-1] = 'Pineapple'; // Add element at the last index of the array
console.log(fruits);
console.log(fruits.length); 

fruits.length = 2;
console.log(fruits);

studentsNames.length = 0; // Remove all elements from the array
console.log(fruits);

Array Methods

Add And Remove From Array

Common methods for modifying arrays:

let studentsNames = ["Amr", "Ahmed", "Beshoy", "Adham", "Khaled", "Shehab"];

console.log(studentsNames);
studentsNames.unshift("Ali"); // Add element at the beginning of the array
console.log(studentsNames);

studentsNames.push("Mohamed"); // Add element at the end of the array
console.log(studentsNames);

studentsNames.push("Amr", "Ahmed", "Beshoy"); // Add multiple elements at the end of the array
console.log(studentsNames);

studentsNames.shift(); // Remove the first element from the array
console.log(studentsNames);

studentsNames.pop(); // Remove the last element from the array
console.log(studentsNames);

let removeFirst = studentsNames.shift();
console.log(studentsNames);
console.log(`The first removed name is: ${removeFirst}`);

let removeLast = studentsNames.pop();
console.log(studentsNames);
console.log(`The last removed name is: ${removeLast}`);

Searching Array

Methods to find elements in arrays:

let students = ["Amr", "Ahmed", "Beshoy", "Adham", "Khaled", "Shehab", "Ali", "Mohamed"];

// indexOf(Search Element, from Index [Optional])
console.log(students.indexOf("Amr")); // Find the index of the element "Amr"
console.log(students.indexOf("Ahmed", 2));

// lastIndexOf(Search Element, from Index [Optional])
console.log(students.lastIndexOf("Beshoy")); // Find the last index of the element "Beshoy"
console.log(students.lastIndexOf("Adham", -2));

// includes(Value to find, from Index [Optional]) => [ES7]
console.log(students.includes("Khaled")); // Check if the array includes the element "Khaled"
console.log(students.includes("Shehab", 3));

Sorting Array

The sort() method sorts elements as strings by default.

let mixedData = [1, 'Hello', true, {name: 'John'}, 5, 90, ['Apple', 'Banana', 'Orange'], -50, '10'];

console.log(mixedData);
console.log(mixedData.sort());
console.log(mixedData.reverse());
console.log(mixedData.sort().reverse());

Slicing Array

slice() extracts a portion of an array without modifying the original.

let numbers = ['Apple', 'Banana', 'Orange', 'Strawberries', 'Pineapple', 'Pomegranate', 'Watermelon'];
console.log(numbers);

// slice(Start [Optional], End [Optional] Not including End)
console.log(numbers.slice());
console.log(numbers.slice(1)); // Get the last elements
console.log(numbers.slice(0, 3)); // Get the first 3 elements
console.log(numbers.slice(-3));
console.log(numbers.slice(1, -2));
console.log(numbers.slice(-4, -2));
console.log(numbers);

// splice(Start [Mandatory], Delete Count [Optional] [0 no remove], The item to Add [Optional])
console.log(numbers.splice(2));
console.log(numbers.splice(2, 2));
console.log(numbers.splice(2, 2, 'Grapes', 'Mango'));
console.log(numbers);

Joining Arrays

Combine arrays or convert them to strings:

let friends = ["Ahmed", "Sayed", "Ali", "Osama", "Gamal"];
let newFriends = ["Samar", "Sarah"];
let schoolFriends = ["Arwa", "Mohammed", "Shrouk"];

let allFriends = friends.concat(newFriends, schoolFriends);
console.log(allFriends);

let allFriends0 = friends.concat(newFriends, schoolFriends, ["Shady", "Hady"]);
console.log(allFriends0);

console.log(allFriends.join());
console.log(allFriends.join(""));
console.log(allFriends.join(" @ "));
console.log(allFriends.join("|"));

Array Challenges

Array Challenge

Create a function that takes an array of numbers and returns:

  1. The sum of all numbers
  2. The average of the numbers
  3. A new array with only even numbers
function processNumbers(numbers) {
    // Sum
    const sum = numbers.reduce((total, num) => total + num, 0);
    
    // Average
    const average = sum / numbers.length;
    
    // Even numbers
    const evens = numbers.filter(num => num % 2 === 0);
    
    return { sum, average, evens };
}

const result = processNumbers([1, 2, 3, 4, 5]);
console.log(result);

Output: { sum: 15, average: 3, evens: [2, 4] }
